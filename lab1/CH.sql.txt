CREATE TABLE IF NOT EXISTS yury_zenin
(
    timestamp UInt64
    referer String,
    location String,
    remoteHost String,
    partyId String,
    sessionId String,
    pageViewId String,
    eventType String,
    item_id String,
    item_price String,
    item_url String,
    basket_price String,
    detectedDuplicate UInt8,
    detectedCorruption UInt8,
    firstInSession UInt8,
    userAgentName String
)
ENGINE = MergeTree()
PARTITION BY timestamp
PRIMARY KEY (timestamp, sessionId)
ORDER BY (timestamp, sessionId)
